# RAF（requestAnimationFrame）

    //判断是否可以使用requestAnimFrame来替换setTimeout
    window.requestAnimFrame =
        window.requestAnimationFrame ||
        window.webkitRequestAnimationFrame ||
        function(callback) {
            return window.setTimeout(callback, 1000 / 30);
        };
